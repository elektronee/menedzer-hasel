cmake_minimum_required(VERSION 3.14)
project(ManagerHasel)
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_library(ManagerHaselLib klasy.cpp klasy.h)
add_executable(ManagerHasel main.cpp)
target_link_libraries(ManagerHasel ManagerHaselLib)

include(FetchContent)
FetchContent_Declare(
  googletest
  URL https://github.com/google/googletest/archive/03597a01ee50ed33e9dfd640b249b4be3799d395.zip
)
set(gtest_force_shared_crt ON CACHE BOOL "" FORCE)
FetchContent_MakeAvailable(googletest)

add_executable(ManagerHaselTests test_klasy.cpp)
target_link_libraries(ManagerHaselTests ManagerHaselLib gtest_main)
include(GoogleTest)
gtest_discover_tests(ManagerHaselTests)
